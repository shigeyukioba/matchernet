[docs]    def update(self, inputs):
        mu = {}
        for key in inputs:
            mu[key] = inputs[key].data["mu"]
        mean = reduce(add, mu.values()) / len(inputs)
        for key in inputs:
            if inputs[key] is not None:
                self.results[key].data["mu"] = (mean - mu[key]) * 0.1
